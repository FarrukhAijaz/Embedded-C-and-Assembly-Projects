int mcount=0;

void initfunc(void){
SYSCTL->RCGCGPIO |= 0x2F;
GPIOB->DIR       |= 0xF0; 
GPIOB->DEN       |= 0xFF; 
GPIOB->AFSEL     &=~0xFF;	
	
	
SysTick->LOAD     = 5000; 
SysTick->CTRL     = 3; 
SysTick->VAL      = 0; 
}

 mcount++;
}

unsigned char getbutton(void){
unsigned char input;

}
